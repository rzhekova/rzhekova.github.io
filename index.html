Temperature Conversion

This Ruby program will prompt the user for a temperature in degrees Celsius and let the user know what the corresponding temperature is in Fahrenheit. 

puts "Enter degrees in Celsius:"
celsius = gets.to_f
	
	def celsius_to_fahrenheit(celsius)
	  	return celsius * 1.8 + 32 
	end
	
puts "The temperature is #{celsius_to_fahrenheit(celsius)}F."

n = gets.chomp.to_i
	
	def calculate_the_ordinal(n)
	  	right_most_digit = n % 10  
	  	if n == 11 || n == 12 || n == 13
	    	return "th"
	  	elsif right_most_digit == 1
	   	 	return "st"
	  	elsif right_most_digit == 2
	    	return "nd"
	  	elsif right_most_digit == 3
	    	return "rd"
	  	else
	    	return "th"
	 	end
	end
	
puts "That's the #{n}" + calculate_the_ordinal(n) + " item!"

Foobar Challenge

This Ruby program is an adaptation of a classic technical interview problem which displays sequences of the Foobar pattern. 

puts "How many items do you want to see?"
number_of_items = gets.chomp.to_i
	
	def foobar(number_input)
	   n = 1
	
	   while n != number_input + 1
	     if n % 15 == 0
	     puts "Foobar"
	    elsif n % 5 == 0
	     puts "Bar"
	    elsif n % 3 == 0
	     puts "Foo"
	    else
	     puts n
	    end
	    
	     n = n + 1
	  end
	end
	
foobar(number_of_items)
